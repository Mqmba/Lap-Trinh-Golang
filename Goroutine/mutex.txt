package main

import (
	"fmt"
	"sync"
)

func main() {
	tokens := 0
	var wg sync.WaitGroup
	var mu sync.Mutex

	for i := 1; i <= 1000; i++ {
		wg.Add(1)
		go func() {
			mu.Lock()
			tokens++    // Khóa tài nguyên
			mu.Unlock() // Tăng counter an toàn
			wg.Done()   // Mở khóa tài nguyên
		}()
	}
	wg.Wait()
	fmt.Println(tokens)
}
